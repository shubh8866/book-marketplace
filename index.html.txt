<script>
// Load saved books
let books = JSON.parse(localStorage.getItem("books")) || [];

function displayBooks() {
    let bookContainer = document.getElementById("books");
    bookContainer.innerHTML = "";

    books.forEach((book, index) => {
        let div = document.createElement("div");
        div.className = "book-card";

        div.innerHTML = `
            <h3>${book.title}</h3>
            <p><b>Author:</b> ${book.author}</p>
            <p><b>Price:</b> ₹${book.price}</p>
            <p><b>Contact:</b> ${book.contact}</p>
            <button class="buy-btn" onclick="buyBook(${index})">Buy</button>
        `;

        bookContainer.appendChild(div);
    });
}

// Add Book
function addBook() {
    let title = document.getElementById("title").value;
    let author = document.getElementById("author").value;
    let price = document.getElementById("price").value;
    let contact = document.getElementById("contact").value;

    if (!title || !author || !price || !contact) {
        alert("Please fill all fields");
        return;
    }

    let book = { title, author, price, contact };
    books.push(book);

    localStorage.setItem("books", JSON.stringify(books));

    alert("Book added successfully!");

    // Clear fields
    document.getElementById("title").value = "";
    document.getElementById("author").value = "";
    document.getElementById("price").value = "";
    document.getElementById("contact").value = "";

    displayBooks();
}

// Buy Book
function buyBook(index) {
    let selectedBook = books[index];

    alert(`Contact seller at: ${selectedBook.contact}`);

    // Remove after purchase (optional)
    books.splice(index, 1);
    localStorage.setItem("books", JSON.stringify(books));

    displayBooks();
}

// Initial load
displayBooks();
</script>
